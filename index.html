calculator.cpp

#include <iostream>
using namespace std;

int main(void){

    int num1;
    int num2;
    
    cout << "Enter Number 1 : "; cin >> num1;
    cout << "Enter Number 2 : "; cin >> num2;

    cout << "Number 1 + Number 2 = " << num1 + num2 << endl;
    cout << "Number 1 * Number 2 = " << num1 * num2 << endl;
    cout << "Number 1 / Number 2 = " << num1 / num2 << endl;
    cout << "Number 1 - Number 2 = " << num1 - num2 << endl;

    return 0;

}
